create or replace 
PROCEDURE "PRC_INSERT_HISTORY_SMS"
(
  P_APPLICATION IN varchar2,
  P_SEQUENCIE IN NUMBER,
  P_INITIAL_SOURCE_ADDR IN NUMBER,
  P_INITIAL_DESTINATION_ADDR IN NUMBER,
  P_PORT_IN IN NUMBER,
  P_PORT_IN_DESCRIPTION IN VARCHAR2,
  P_PORT_OUT IN NUMBER,
  P_SHORT_MESSAGE IN VARCHAR2,
  P_MESSAGE_ID IN VARCHAR2,
  P_STATUS IN NUMBER,
  P_LAST_UPDATE_TIME IN DATE,
  P_HISTORY_INSERT_TIME IN DATE,
  P_SOURCE_TABLE_NAME IN VARCHAR2,
  P_USUARIO IN VARCHAR2
)AS 
BEGIN
 INSERT INTO PRUEBA(APPLICATION, SEQUENCIE, INITIAL_SOURCE_ADDR, PORT_IN, PORT_IN_DESCRIPTION, PORT_OUT, SHORT_MESSAGE, MESSAGE_ID,
 STATUS, LAST_UPDATE_TIME,HISTORY_INSERT_TIME, SOURCE_TABLE_NAME,USUARIO) VALUES (P_APPLICATION, P_SEQUENCIE, P_INITIAL_SOURCE_ADDR,
 P_PORT_IN, P_PORT_IN_DESCRIPTION, P_PORT_OUT, P_SHORT_MESSAGE, P_MESSAGE_ID, P_STATUS, P_LAST_UPDATE_TIME, P_HISTORY_INSERT_TIME,P_SOURCE_TABLE_NAME,P_USUARIO);
END PRC_INSERT_HISTORY_SMS;
